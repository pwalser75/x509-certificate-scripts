Create CA certificate
keytool -genkeypair -keyalg {{settings.keyAlg}} -keysize {{settings.keySize}} -alias {{settings.alias}} -keystore {{settings.keystore}} -dname "{{settings.dname}}" -storepass {{settings.keystorePass}} -keypass {{settings.keystoreKeyPass}} -ext KeyUsage:critical="keyCertSign,cRLSign" -ext BasicConstraints:critical="ca:true,pathlen:1" -validity {{settings.validityDays}}
Export CA public certificate
keytool -export -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -keypass {{settings.keystoreKeyPass}} -alias {{settings.alias}} -file {{settings.alias}}.cer keytool -exportcert -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -keypass {{settings.keystoreKeyPass}} -alias {{settings.alias}} -rfc -file {{settings.alias}}.pem
Create Server certificate
keytool -genkey -alias {{settings.alias}} -keyalg {{settings.keyAlg}} -keysize {{settings.keySize}} -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -validity {{settings.validityDays}} -ext KeyUsage:critical="digitalSignature,keyEncipherment" -ext EKU="{{settings.eku}}" -keypass {{settings.keystoreKeyPass}} -dname "{{settings.dname}}"
Create CSR (certificate signing request)
keytool -certreq -alias {{settings.alias}} -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -keypass {{settings.keystoreKeyPass}} -file {{settings.alias}}.csr
CA: Sign CSR (certificate signing request)
keytool -gencert -keystore {{settings.caKeystore}} -storepass {{settings.caKeystorePass}} -alias {{settings.caAlias}} -validity {{settings.validityDays}} -ext KeyUsage:critical="digitalSignature,keyEncipherment" -ext EKU="{{settings.eku}}" -ext SAN={{settings.serverNames}} -keypass {{settings.caKeystoreKeyPass}} -infile {{settings.alias}}.csr -outfile {{settings.alias}}-signed.cer
Import CA cert and signed CSR (certificate signing request)
keytool -import -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -alias {{settings.caAlias}} -file {{settings.caAlias}}.cer -noprompt keytool -import -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -keypass {{settings.keystorePass}} -alias {{settings.alias}} -file {{settings.alias}}-signed.cer -noprompt
Export server keystore as PKCS12
keytool -importkeystore -srckeystore {{settings.keystore}} -srcstorepass {{settings.keystorePass}} -alias {{settings.alias}} -srckeypass {{settings.keystorePass}} -destkeystore {{settings.keystore}}.p12 -deststoretype PKCS12 -deststorepass {{settings.keystorePass}} -destkeypass {{settings.keystorePass}}
Export server public certificate
keytool -export -keystore {{settings.keystore}} -storepass {{settings.keystorePass}} -alias {{settings.alias}} -rfc -file {{settings.alias}}.cer
Create Server truststore
keytool -keystore {{settings.truststore}} -storepass {{settings.truststorePass}} -import -file {{settings.caAlias}}.cer -alias {{settings.caAlias}} -noprompt keytool -importkeystore -srckeystore {{settings.truststore}} -srcstorepass {{settings.truststorePass}} -destkeystore {{settings.alias}}.p12 -srcstoretype JKS -deststoretype PKCS12 -deststorepass {{settings.truststorePass}}
